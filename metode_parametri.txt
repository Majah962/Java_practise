
class Robot {
	//možemo i tako dobiti tekst,znaèi ovdje kod speak uneesemo STRING "nešto" -to se zove PARAMETERS i to isto 
	//kod sysout moramo imati isto to "nešto"
	public void speak(String text) {
		System.out.println(text);
	}
	
	public void jump(int height) {
		System.out.println("Jumping: " + height);
	}
	
	public void move(String direction, double distance) {
		System.out.println("Moving " + distance + " metres in direction " + direction);
	}
}



public class Application {

	public static void main(String[] args) {
		
		Robot sam = new Robot();
		//ako ovdje napišemo neki tekst moramo ga gore popraviti
		sam.speak("Hi I'm Sam.");
		sam.jump(7);
		sam.move("West", 12.2);
		
		String greeting = "Hello there.";
		sam.speak(greeting);
		
		int value = 14;
		sam.jump(value);

	}

}
